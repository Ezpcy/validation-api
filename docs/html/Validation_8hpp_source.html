75 inline void extractNullOptions(const pugi::xml_node &doc) {
76 // Store the name
77 std::string forField = std::string(doc.name());
78 int currentKey = 0;
79
80 for (pugi::xml_node node : doc.children()) {
81 // Recursive call for child nodes
82 extractNullOptions(node);
83
84 if (std::string(node.name()) == "AllowNullIf") {
85 for (pugi::xml_node nullId : node.children()) {
86 // Iterate over attributes of the current <Null> child
87 for (pugi::xml_attribute_iterator attr = nullId.attributes_begin();
88 attr != nullId.attributes_end(); ++attr) {
89 // Check if the UUID is valid
90 if (isValidUuid(std::string(attr->value()))) {
91
92 auto it = nullOptions_.find(forField);
93
94 // If the key is not found, insert it with a new vector
95 if (it == nullOptions_.end()) {
96 nullOptions_[forField].insert(
97 {currentKey, {{nullId.name(), attr->value()}}});
98
99 } else {
100 // If the key is found, append the new value to the vector
101 nullOptions_[forField][currentKey].push_back(
102 {nullId.name(), attr->value()});
103 }
104 } else {
105 // If the UUID is invalid, log an error
106 errors_.push_back(
107 {std::format("XML Null option: "),
108 std::format("Field {} skipped because of invalid Uuid.",
109 std::string(nullId.name()))});
110 }
111 }
112 }
113 currentKey++;
114 } else {
115 continue;

120 /*
121 * @brief findJsonField
122 * @param jsonObj nodeName
123 * @details Recursive function to find a field in a JSON object
124 * @return std::pair<bool, nlohmann::json> We need return a pair, in case the
125 * key exists but the value is a nullptr
126 */
127 inline std::pair<bool, nlohmann::json>
128 findJsonField(const nlohmann::json &jsonObj, const std::string &nodeName) {
129 // If the key exists at the current level, return it
130 if (jsonObj.contains(nodeName)) {
131 return std::pair(true, jsonObj[nodeName]);
132 }
133
134 // Otherwise, look through all objects in the current level
135 for (auto &[key, value] : jsonObj.items()) {
136 if (value.is_object() || value.is_array()) {
137 // Recursively search in nested JSON objects
138 nlohmann::json result = findJsonField(value, nodeName);
139 if (result != std::pair(false, nullptr)) {
140 return result;
141 }
142 }
143 }
144
145 // Return null if the key was not found
146 return std::pair(false, nullptr);
147 }
