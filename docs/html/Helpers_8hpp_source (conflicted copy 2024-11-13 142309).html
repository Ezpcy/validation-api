54inline bool validateAhv(std::string &val) {
55 // AHV regex
56 const std::regex ahv_regex(R"(^[7][5][6][.][\d]{4}[.][\d]{4}[.][\d]{2}$)");
57
58 // Match the val with the regex
59 if (!std::regex_match(val, ahv_regex)) {
60 return false;
61 }
62
63 // Get the last char of the parameter
64 char lastChar = val.back();
65
66 // Check if it's a digit
67 if (!std::isdigit(lastChar)) {
68 return false;
69 }
70
71 // Cast the last digit as an integer
72 int lastDigit = lastChar - '0';
73
74 // Remove the last digit from the original string
75 val.pop_back();
76
77 // Remove dots from the string
78 val.erase(std::remove(val.begin(), val.end(), '.'), val.end());
79
80 // Reverse the string for the next part of the calculation
81 std::reverse(val.begin(), val.end());
82
83 // Calculate the sum by multiplying every second digit by 3
84 int sum = 0;
85 for (std::size_t i = 0; i < val.size(); ++i) {
86 char c = val[i];
87
88 // Make sure we only work with digits
89 if (!std::isdigit(c)) {
90 return false;
91 }
92
93 // Convert char to int
94 int digit = c - '0';
95
96 // Multiply every second digit by 3
97 int add = (i % 2 == 0) ? digit * 3 : digit;
98
99 sum += add;
100 }
101
102 // Calculate the next ten by rounding up the sum divided by 10
103 int nextTen = static_cast<int>(std::ceil(sum / 10.0) * 10);
104
105 // Calculate the checksum by subtracting the sum from the next ten
106 int checkSum = nextTen - sum;
107
108 // Compare the checksum with the original last digit
109 if (checkSum != lastDigit) {
110 return false;
111 }
112
113 return true;
114}

123inline bool validateIban(const std::string &val) {
124 // Remove all whitespaces from the IBAN
125 std::string address;
126 for (char ch : val) {
127 if (!std::isspace(static_cast<unsigned char>(ch))) {
128 address += ch;
129 }
130 }
131
132 // Check if the IBAN is 21 characters long
133 if (address.size() != 21) {
134 return false;
135 }
136
137 // Check if the IBAN starts with "CH"
138 if (address.substr(0, 2) != "CH") {
139 return false;
140 }
141
142 // Move the first four characters to the back
143 std::string rearranged = address.substr(4) + address.substr(0, 4);
144
145 // Convert the rearranged string into a numerical form (digits where letters
146 // A-Z are replaced by 10-35)
147 std::string numerical;
148 for (char ch : rearranged) {
149 if (std::isdigit(ch)) {
150 numerical += ch;
151 } else if (std::isalpha(ch)) {
152 // Convert letters 'A'-'Z' to numbers '10'-'35'
153 numerical += std::to_string(ch - 'A' + 10);
154 } else {
155 return false; // Invalid character
156 }
157 }
158
159 // Calculate the checksum using modulo 97
160 const int MOD = 97;
161 unsigned long long checksum = 0;
162 for (char digit : numerical) {
163 // Multiply the current checksum by 10 and add the next digit
164 checksum = (checksum * 10 + (digit - '0')) % MOD;
165 }
166
167 // The IBAN is valid if the result of modulo 97 is 1
168 if (checksum != 1) {
169 return false;
170 }
171
172 // Additionally check that the digits between index 2 and 4 are not "00" or
173 // "01"
174 std::string checkDigits = address.substr(2, 2);
175 if (checkDigits == "00" || checkDigits == "01") {
176 return false;
177 }
178
179 return true;
180}
