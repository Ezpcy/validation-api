66inline bool validateAhv(std::string &val) {
67 // AHV regex
68 const std::regex ahv_regex(R"(^[7][5][6][.][\d]{4}[.][\d]{4}[.][\d]{2}$)");
69
70 // Match the val with the regex
71 if (!std::regex_match(val, ahv_regex)) {
72 return false;
73 }
74
75 // Get the last char of the parameter
76 char lastChar = val.back();
77
78 // Check if it's a digit
79 if (!std::isdigit(lastChar)) {
80 return false;
81 }
82
83 // Cast the last digit as an integer
84 int lastDigit = lastChar - '0';
85
86 // Remove the last digit from the original string
87 val.pop_back();
88
89 // Remove dots from the string
90 val.erase(std::remove(val.begin(), val.end(), '.'), val.end());
91
92 // Reverse the string for the next part of the calculation
93 std::reverse(val.begin(), val.end());
94
95 // Calculate the sum by multiplying every second digit by 3
96 int sum = 0;
97 for (std::size_t i = 0; i < val.size(); ++i) {
98 char c = val[i];
99
100 // Make sure we only work with digits
101 if (!std::isdigit(c)) {
102 return false;
103 }
104
105 // Convert char to int
106 int digit = c - '0';
107
108 // Multiply every second digit by 3
109 int add = (i % 2 == 0) ? digit * 3 : digit;
110
111 sum += add;
112 }
113
114 // Calculate the next ten by rounding up the sum divided by 10
115 int nextTen = static_cast<int>(std::ceil(sum / 10.0) * 10);
116
117 // Calculate the checksum by subtracting the sum from the next ten
118 int checkSum = nextTen - sum;
119
120 // Compare the checksum with the original last digit
121 if (checkSum != lastDigit) {
122 return false;
123 }
124
125 return true;
126}

135inline bool validateIban(const std::string &val) {
136 // Remove all whitespaces from the IBAN
137 std::string address;
138 for (char ch : val) {
139 if (!std::isspace(static_cast<unsigned char>(ch))) {
140 address += ch;
141 }
142 }
143
144 // Check if the IBAN is 21 characters long
145 if (address.size() != 21) {
146 return false;
147 }
148
149 // Check if the IBAN starts with "CH"
150 if (address.substr(0, 2) != "CH") {
151 return false;
152 }
153
154 // Move the first four characters to the back
155 std::string rearranged = address.substr(4) + address.substr(0, 4);
156
157 // Convert the rearranged string into a numerical form (digits where letters
158 // A-Z are replaced by 10-35)
159 std::string numerical;
160 for (char ch : rearranged) {
161 if (std::isdigit(ch)) {
162 numerical += ch;
163 } else if (std::isalpha(ch)) {
164 // Convert letters 'A'-'Z' to numbers '10'-'35'
165 numerical += std::to_string(ch - 'A' + 10);
166 } else {
167 return false; // Invalid character
168 }
169 }
170
171 // Calculate the checksum using modulo 97
172 const int MOD = 97;
173 unsigned long long checksum = 0;
174 for (char digit : numerical) {
175 // Multiply the current checksum by 10 and add the next digit
176 checksum = (checksum * 10 + (digit - '0')) % MOD;
177 }
178
179 // The IBAN is valid if the result of modulo 97 is 1
180 if (checksum != 1) {
181 return false;
182 }
183
184 // Additionally check that the digits between index 2 and 4 are not "00" or
185 // "01"
186 std::string checkDigits = address.substr(2, 2);
187 if (checkDigits == "00" || checkDigits == "01") {
188 return false;
189 }
190
191 return true;
192}
